// スクロールに合わせて中央の画像を明るくする処理 const carousel = document.querySelector('.video-carousel'); const items = document.querySelectorAll('.video-item'); carousel.addEventListener('scroll', () => { let minDistance = Infinity; let activeItem = null; items.forEach(item => { const rect = item.getBoundingClientRect(); const distance = Math.abs((rect.left + rect.right) / 2 - window.innerWidth / 2); if (distance < minDistance) { minDistance=distance; activeItem=item; } }); items.forEach(item=> item.classList.remove('is-active')); if (activeItem) activeItem.classList.add('is-active'); }); // 初期状態でも中央を明るくするために一度発火させる carousel.dispatchEvent(new Event('scroll'));
